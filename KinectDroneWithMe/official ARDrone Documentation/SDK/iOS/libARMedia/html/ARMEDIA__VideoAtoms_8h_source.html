73 /* Atoms :
74 LEGEND :
75 -> specific = specific function to generate this atom
76 -> empty = use atomFromData (0, "name", NULL);
77 -> from data = use atomFromData (dataSize, "name", dataPointer);
78 
79 ftyp -> specific
80 pvat -> specific
81 mdat -> specific (include freeAtom if needed)
82 moov -> empty
83  |- mvhd -> specific
84  |- trak -> empty
85  | |- tkhd -> specific
86  | \- mdia -> empty
87  | |- mdhd -> specific
88  | |- hdlr -> specific (for mdia)
89  | \- minf -> empty
90  | |- vmhd -> specific
91  | |- hdlr -> specific (for minf)
92  | |- dinf -> empty
93  | | \- dref -> specific
94  | \- stbl -> empty
95  | |- stsd -> specific
96  | |- stts -> from data
97  | |- stss -> from data (i frame positions as uint32_t network endian)
98  | |- stsc -> specific
99  | |- stsz -> from data (frames sizes as uint32_t network endian)
100  | \- stco -> from data (frames offset as uint32_t network endian)
